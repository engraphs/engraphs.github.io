4_A_Voyager
The statement is recorded in electronic impulses which can be converted into printed words . 



% 4
arc([w('is',4,3),w('recorded',4,4)],[w('statement',4,2)],['arg1']).
arc([w('is',4,3),w('recorded',4,4)],[w('impulses',4,7)],['clr','loc',w('in',4,5)]).
arc([w('impulses',4,7)],[w('electronic',4,6)],['attribute']).
arc([w('impulses',4,7)],[w('be',4,10),w('converted',4,11)],['arg1_of']).
arc([w('be',4,10),w('converted',4,11)],[w('can',4,9)],['md']).
arc([w('be',4,10),w('converted',4,11)],[w('words',4,14)],['clr','dir',w('into',4,12)]).
arc([w('words',4,14)],[w('printed',4,13)],['arg1_of']).




( (IP-MAT (NP-SBJ (D The)
                  (N statement))
          (BEP;~cat_Ve is)
          (IP-PPL-CAT (NP-LGS *)
                      (VVN;~Tn.pr recorded)
                      (PP-CLR-LOC (P-ROLE in)
                                  (NP (ADJP (ADJ electronic))
                                      (NS impulses)
                                      (IP-REL (NP-SBJ (RPRO which))
                                              (MD can)
                                              (BE;~cat_Ve be)
                                              (IP-PPL-CAT (NP-LGS *)
                                                          (VVN;~Tn.pr converted)
                                                          (PP-CLR-DIR (P-ROLE into)
                                                                      (NP (IP-PPL (NP-LGS *)
                                                                                  (VVN;~Tn printed))
                                                                          (NS words))))))))
          (PUNC .))
  (ID 4_A_Voyager))
